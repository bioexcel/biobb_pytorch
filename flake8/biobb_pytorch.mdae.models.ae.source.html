22 class AutoEncoder(BaseCV, pl.LightningModule):
23 """AutoEncoding Collective Variable.
24 It is composed by a first neural network (encoder) which projects
25 the input data into a latent space (the CVs). Then a second network (decoder) takes
26 the CVs and tries to reconstruct the input data based on them. It is an unsupervised learning approach,
27 typically used when no labels are available. This CV is inspired by [1]_.
28  
29 Furthermore, it can also be used lo learn a representation which can be used not to reconstruct the data but
30 to predict, e.g. future configurations.
31  
32 **Data**: for training it requires a DictDataset with the key 'data' and optionally 'weights' to reweight the
33 data as done in [2]_. If a 'target' key is present this will be used as reference for the output of the decoder,
34 otherway this will be compared with the input 'data'. This feature can be used to train a time-lagged autoencoder [3]_
35 where the task is not to reconstruct the input but the output at a later step.
36  
37 **Loss**: reconstruction loss (MSELoss)
38  
39 References
40 ----------
41 .. [1] W. Chen and A. L. Ferguson, “ Molecular enhanced sampling with autoencoders: On-the-fly collective
42 variable discovery and accelerated free energy landscape exploration,” JCC 39, 2079–2102 (2018)
43 .. [2] Z. Belkacemi, P. Gkeka, T. Lelièvre, and G. Stoltz, “ Chasing collective variables using autoencoders and biased
44 trajectories,” JCTC 18, 59–78 (2022)
45 .. [3] C. Wehmeyer and F. Noé, “Time-lagged autoencoders: Deep learning of slow collective variables for molecular
46 kinetics,” JCP 148, 241703 (2018).
47  
48 See also
49 --------
50 mlcolvar.core.loss.MSELoss
51 (weighted) Mean Squared Error (MSE) loss function.
52 """

56 def __init__(
57 self,
58 n_features: int,
59 n_cvs: int,
60 encoder_layers: list,
61 decoder_layers: list = None,
62 options: dict = None,
63 **kwargs,
64 ):
65 """
66 Define a CV defined as the output layer of the encoder of an autoencoder model (latent space).
67 The decoder part is used only during the training for the reconstruction loss.
68 By default a module standardizing the inputs is also used.
69  
70 Parameters
71 ----------
72 encoder_layers : list
73 Number of neurons per layer of the encoder
74 decoder_layers : list, optional
75 Number of neurons per layer of the decoder, by default None
76 If not set it takes automaically the reversed architecture of the encoder
77 options : dict[str,Any], optional
78 Options for the building blocks of the model, by default None.
79 Available blocks: ['norm_in', 'encoder','decoder'].
80 Set 'block_name' = None or False to turn off that block
81 """
82 super().__init__(
83 in_features=n_features, out_features=n_cvs, **kwargs
84 )
85  
86 self.save_hyperparameters('n_features', 'n_cvs', 'encoder_layers', 'decoder_layers', 'options')
87  
88 # ======= LOSS =======
89 # Reconstruction (MSE) loss
90 self.loss_fn = MSELoss()
91  
92 # ======= OPTIONS =======
93 # parse and sanitize
94 options = self.parse_options(options)
95  
96 # if decoder is not given reverse the encoder
97 if decoder_layers is None:
98 decoder_layers = encoder_layers[::-1]
99  
100 # ======= BLOCKS =======
101  
102 # initialize norm_in
103 o = "norm_in"
104 if (options[o] is not False) and (options[o] is not None):
105 self.norm_in = Normalization(self.in_features, **options[o])
106  
107 # initialize encoder
108 o = "encoder"
109 self.encoder = FeedForward([n_features] + encoder_layers + [n_cvs], **options[o])
110  
111 # initialize decoder
112 o = "decoder"
113 self.decoder = FeedForward([n_cvs] + decoder_layers + [n_features], **options[o])
114  
115 self.eval_variables = ["xhat", "z"]
116  
117 def forward_cv(self, x: torch.Tensor) -> torch.Tensor:
118 """Evaluate the CV without pre or post/processing modules."""
119 if self.norm_in is not None:
120 x = self.norm_in(x)
121 x = self.encoder(x)
122 return x
123  
124 def decode(self, z: torch.Tensor) -> torch.Tensor:
125 """Decode the latent space into the original input space."""
126 x = self.decoder(z)
127 if self.norm_in is not None:
128 x = self.norm_in.inverse(x)
129 return x
130  
131 def encode_decode(self, x: torch.Tensor) -> torch.Tensor:
132 """Pass the inputs through both the encoder and the decoder networks."""
133 x = self.forward_cv(x)
134 x = self.decoder(x)
135 if self.norm_in is not None:
136 x = self.norm_in.inverse(x)
137 return x
138  
139 def evaluate_model(self, batch, batch_idx=None):
140 """Evaluate the model on the data, computing the reconstruction loss."""
141  
142 x = batch['data']
143 z = self.forward_cv(x)
144 x_hat = self.decoder(z)
145  
146 if self.norm_in is not None:
147 x_hat = self.norm_in.inverse(x_hat)
148  
149 return x_hat, z
150  
151 def training_step(self, train_batch, batch_idx):
152 """Compute and return the training loss and record metrics."""
153 # =================get data===================
154 x = train_batch["data"]
155 loss_kwargs = {}
156 if "weights" in train_batch:
157 loss_kwargs["weights"] = train_batch["weights"]
158 # =================forward====================
159 x_hat = self.encode_decode(x)
160 # ===================loss=====================
161 if "target" in train_batch:
162 x_ref = train_batch["target"]
163 else:
164 x_ref = x
165  
166 # if self.norm_in is not None:
167 # x_ref = self.norm_in(x_ref)
168  
169 loss = self.loss_fn(x_hat, x_ref, **loss_kwargs)
170  
171 # ====================log=====================
172 name = "train" if self.training else "valid"
173 self.log(f"{name}_loss", loss, on_epoch=True, on_step=True, prog_bar=True, logger=True)
174 return loss
175  
176 def get_decoder(self, return_normalization=False):
177 """Return a torch model with the decoder and optionally the normalization inverse"""
178 if return_normalization:
179 if self.norm_in is not None:
180 inv_norm = Inverse(module=self.norm_in)
181 decoder_model = torch.nn.Sequential(*[self.decoder, inv_norm])
182 else:
183 raise ValueError(
184 "return_normalization is set to True but self.norm_in is None"
185 )
186 else:
187 decoder_model = self.decoder
188 return decoder_model
