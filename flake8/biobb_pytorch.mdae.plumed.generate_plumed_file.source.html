6 def parse_ndx(ndx_path, group_name):
7 atoms = []
8 with open(ndx_path, 'r') as f:
9 in_group = False
10 for line in f:
11 line = line.strip()
12 if line.startswith('[') and line.endswith(']'):
13 current_group = line[1:-1].strip()
14 in_group = (current_group == group_name)
15 elif in_group and line:
16 atoms.extend(int(x) for x in line.split())
